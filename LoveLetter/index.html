Boy i = new Boy("汤包");
Girl u = new Girl("肉包");
// 2022年12月19日，汤包向肉包表白了
i.love(u);
// 幸运的是，肉包接受了
u.accepted();
// 12月20日，它们正式在一起了
// 它们兴奋的分享了这个消息，所有人都为他们送上了祝福
wait(25*60*60);
fall_in_love(i, u);
// 然后呢，汤包每天都在想着肉包
i.miss(u);
// 他非常的在意，自己对肉包够不够好
i.takeCareOf(u);
// 但是，汤包总是有事情要干
i.setstatus(_BUSY_); // 汤包很难受，他很想陪肉包多玩一会
while (i.status() == _BUSY_) {
bool check = i.work();
// 所以它干啊干，希望能把事情干完
if ( !check ) i.setstatus(_IDLE_);
}
// 因为他真的，想和你去很多很多地方
while ( TRUE ) {
travel(i, u);
}
